        def eq?(*argv,&block)
                hash=argv.pop
                objects=argv.flatten
                result=true
                
                objects.each {|_object|
                        hash.each {|_property,_operand|
                                match=Splattr::Helpers::actual_result(self,_object,_property,_operand)
                                result_object = Result.new :eq, self, _object, _property, _operand, match

                                if block_given?
                                        yield self, result_object
                                end

                                result = false if result && !match
                        }
                }

                result
        end
